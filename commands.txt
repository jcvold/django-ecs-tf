# Push Image to ECR 
aws ecr get-login-password --region us-east-2 | docker login --username AWS --password-stdin 559717033685.dkr.ecr.us-east-2.amazonaws.com #Login if necessary
docker push 559717033685.dkr.ecr.us-east-2.amazonaws.com/django-app:latest

# Update Deployment via Script
cd deployment
python update-ecs.py --cluster=sandbox-cluster --service=sandbox-service #requires AWS credentials set as command line variables
